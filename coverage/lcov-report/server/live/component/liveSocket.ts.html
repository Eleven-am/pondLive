import {PondChannel, Channel} from "../../socket";
import {Subscription} from "../../utils";
import {ComponentManager} from "../componentManager";
import {LiveRouter} from "./liveRouter";
 
export class LiveSocket<LiveContext extends Object> {
    private _liveContext: LiveContext;
    public readonly clientId: string;
    private readonly _pond: PondChannel;
    private _channel: Channel | null;
    private readonly _manager: ComponentManager;
    private readonly _subscriptions: {name: string, sub: Subscription}[];
 
    constructor(clientId: string, pond: PondChannel, manager: ComponentManager) {
        this._liveContext = {} as LiveContext;
        this.clientId = clientId;
        this._pond = pond;
        this._subscriptions = [];
        this._manager = manager;
        this._channel = null;
    }
 
    set channel(channel: Channel) {
        this._channel = channel;
    }
 
    getChannel(name: string) {
        return this._pond.findChannel(c => c.name === name);
    }
 
    get context(): LiveContext {
        return this._liveContext;
    }
 
    assign(assigns: Partial<LiveContext>): void {
        this._liveContext = Object.assign(this._liveContext, assigns);
    }
 
    assignToChannel<AssignData extends Object>(name: string, assigns: AssignData): void {
        const channel = this._pond.findChannel(c => c.name === name);
        if (channel)
            channel.data = assigns;
    }
 
    broadcast<BroadcastData>(channel: string, event: string, data: BroadcastData): void {
        this._pond.broadcastToChannel(channel, this.clientId, {event, data});
    }
 
    getChannelData<AssignData>(name: string): AssignData | null {
        const channel = this._pond.findChannel(c => c.name === name);
        if (channel)
            return channel.data as AssignData;
 
        return null;
    }
 
    subscribe(name: string): void {
        const sub = this._pond.subscribe(name,data => {
            if (this._channel && data.event !== this.clientId) {
                const clientId  = Object.keys(this._channel.info.assigns)[0];
                if (clientId) {
                    const response = this._channel.createPondResponse(clientId);
                    const router = new LiveRouter(response);
                    const info = data.payload;
                    this._manager.handleInfo(info, this, router, response);
                }
            }
        });
 
        this._subscriptions.push({name: name, sub: sub});
    }
 
    unsubscribe(name: string): void {
        const find = this._subscriptions.findIndex(s => s.name === name);
        if (find !== -1) {
            this._subscriptions[find].sub.unsubscribe();
            this._subscriptions.splice(find, 1);
        }
    }
}
